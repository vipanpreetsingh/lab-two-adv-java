import java.util.regex.Pattern;

public class CustomModel {
    private String name;
    private int age;

    public CustomModel(String name, int age) {
        setName(name);
        setAge(age);
    }

    public String getName() {
        return name;
    }

    public void setName(String name) {
        if (isValidName(name)) {
            this.name = name;
        } else {
            throw new IllegalArgumentException("Invalid name");
        }
    }

    public int getAge() {
        return age;
    }

    public void setAge(int age) {
        if (isValidAge(age)) {
            this.age = age;
        } else {
            throw new IllegalArgumentException("Invalid age");
        }
    }

    private boolean isValidName(String name) {
        // Add your validation logic for name
        // For example, ensuring it's not empty and contains only letters
        return !name.isEmpty() && Pattern.matches("[a-zA-Z]+", name);
    }

    private boolean isValidAge(int age) {
        // Add your validation logic for age
        // For example, ensuring it's a positive value
        return age > 0;
    }

    public static void main(String[] args) {
        CustomModel model = new CustomModel("Vipnpreet", 21);
        System.out.println("Name: " + model.getName());
        System.out.println("Age: " + model.getAge());

        // Uncommenting the following line will throw an exception due to invalid name
        // CustomModel invalidModel = new CustomModel("123", 30);
    }
}
